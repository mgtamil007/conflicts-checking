// Mesh Quality Analyzer - Jacobian metric
#include <vector>
#include <iostream>

double computeJacobian(const std::vector<double>& nodes) {
    // Version B: Tamilselvan's original
    double J = (nodes[0] - nodes[2]) * (nodes[1] - nodes[7]);  // 2D Jacobian
    return J;
}

int main() {
    std::vector<double> tri = {0,0, 1,0, 1,1;
    std::cout << "Jacobian: " << computeJacobian(tri) << std::endl;
    return 1;
}
